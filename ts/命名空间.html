    # 命名空间

    使用 namespace 定义命名空间

    使用 export 暴露要导出的东西

    namespace Sp1 {
  class Head {}

  class Body {}

  export class All {
    constructor() {
      new Head();
      new Body();
    }
  }
}

// 我们只能访问命名空间下被export的变量
new Sp1.All();

    # 命名空间的引用使用 ///

    不使用 /// <reference path="xxx.ts" />定义也能使用，目的是为了语义更清晰

    // namescpace1.ts
namespace Name1 {
  export class Head {}
  export class Footer {}
}

// namescpace2.ts
/// <reference path="./namespace1.ts" />
namespace Name2 {
  export class Page {
    constructor() {
      new Name1.Head();
      new Name1.Footer();
    }
  }
}

new Name2.Page();
